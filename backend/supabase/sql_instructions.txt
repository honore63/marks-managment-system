ALTER TABLE assessments ADD COLUMN IF NOT EXISTS abbr TEXT;

-- Teacher profile fields
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS sdms_code TEXT;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS phone TEXT;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS signature TEXT;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS is_class_teacher BOOLEAN DEFAULT false;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS is_subject_teacher BOOLEAN DEFAULT false;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS temp_password_active BOOLEAN DEFAULT false;

-- Teacher Assignments join table
CREATE TABLE IF NOT EXISTS teacher_assignments (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    teacher_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
    class_id UUID REFERENCES classes(id) ON DELETE CASCADE,
    subject_id UUID REFERENCES subjects(id) ON DELETE SET NULL,
    type TEXT CHECK (type IN ('class', 'subject')),
    created_at TIMESTAMPTZ DEFAULT now()
);

-- Settings table
CREATE TABLE IF NOT EXISTS settings (
    key TEXT PRIMARY KEY,
    value JSONB,
    updated_at TIMESTAMPTZ DEFAULT now()
);

-- ============================================================
-- SECTION 6: DIAGNOSTIC QUERIES (Check your data)
-- Run these to see what is currently in the database
-- ============================================================

-- See all teachers and their assignment count
SELECT p.full_name, p.email, p.role, 
       COUNT(ta.id) AS assignment_count
FROM profiles p
LEFT JOIN teacher_assignments ta ON ta.teacher_id = p.id
WHERE p.role = 'teacher'
GROUP BY p.id, p.full_name, p.email, p.role;

-- See all assignments with teacher + class + subject names
SELECT 
    p.full_name AS teacher,
    c.name AS class,
    s.name AS subject,
    ta.type
FROM teacher_assignments ta
JOIN profiles p ON p.id = ta.teacher_id
JOIN classes  c ON c.id = ta.class_id
LEFT JOIN subjects s ON s.id = ta.subject_id
ORDER BY p.full_name, ta.type;
